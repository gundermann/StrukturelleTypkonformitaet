        fp, new Date() );
    assertThat( alleFreigegebenenFPs, nullValue() );
  }

  @Override
  public void addTriedMethodCall( Method method ) {
    calledMethods.add( method );
  }

  @Override
  public Collection<Method> getTriedMethodCalls() {

  }

}
